q1 = df['sales'].quantile(0.4)
q2 = df['sales'].quantile(0.6)

df = df[(df['sales'] <= q1) | (df['sales']>=q2)]

df['sales_cat'] = df['sales'].apply(
    lambda x: 'Low' if x<=q1 else 'High'
)
